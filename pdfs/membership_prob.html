from scipy.stats import multivariate_normal
def pdf_xdgmm(inputs,alpha,components,V,mu):
    
    output=0
    for i in range(0,components):
        mean=mu[i]
        cov=V[i]
        output=output+alpha[i]*multivariate_normal.pdf(inputs,mean=mean,cov=cov)
    return output
